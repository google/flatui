90 /// @class FontBufferParameters
91 ///
92 /// @brief This class that includes font buffer parameters. It is used as a key
93 /// in the unordered_map to look up FontBuffer.
94 class FontBufferParameters {
95  public:
96  /// @brief The default constructor for an empty FontBufferParameters.
97  FontBufferParameters()
98  : font_id_(kNullHash),
99  text_id_(kNullHash),
100  font_size_(0),
101  size_(mathfu::kZeros2i),
102  caret_info_(false) {}
103 
104  /// @brief Constructor for a FontBufferParameters.
105  ///
106  /// @param[in] font_id The HashedId for the font.
107  /// @param[in] text_id The HashedID for the text.
108  /// @param[in] font_size A float representing the size of the font.
109  /// @param[in] size The size of the FontBuffer.
110  /// @param[in] caret_info A bool determining if the font buffer contains caret
111  /// info.
112  FontBufferParameters(const HashedId font_id, const HashedId text_id,
113  float font_size, const mathfu::vec2i &size,
114  bool caret_info) {
115  font_id_ = font_id;
116  text_id_ = text_id;
117  font_size_ = font_size;
118  size_ = size;
119  caret_info_ = caret_info;
120  }
121 
122  /// @brief The equal-to operator for comparing FontBufferParameters for
123  /// equality.
124  ///
125  /// @param[in] other The other FontBufferParameters to check against for
126  /// equality.
127  ///
128  /// @return Returns `true` if the two FontBufferParameters are equal.
129  /// Otherwise it returns `false`.
130  bool operator==(const FontBufferParameters &other) const {
131  return (font_id_ == other.font_id_ && text_id_ == other.text_id_ &&
132  font_size_ == other.font_size_ && size_.x() == other.size_.x() &&
133  size_.y() == other.size_.y() && caret_info_ == other.caret_info_);
134  }
135 
136  /// @brief The hash function for FontBufferParameters.
137  ///
138  /// @param[in] key A FontBufferParameters to use as the key for
139  /// hashing.
140  ///
141  /// @return Returns a `size_t` of the hash of the FontBufferParameters.
142  size_t operator()(const FontBufferParameters &key) const {
143  // Note that font_id_ and text_id_ are already hashed values.
144  size_t value = (font_id_ ^ (text_id_ << 1)) >> 1;
145  value = value ^ (std::hash<float>()(key.font_size_) << 1) >> 1;
146  value = value ^ (std::hash<bool>()(key.caret_info_) << 1) >> 1;
147  value = value ^ (std::hash<int32_t>()(key.size_.x()) << 1) >> 1;
148  value = value ^ (std::hash<int32_t>()(key.size_.y()) << 1) >> 1;
149  return value;
150  }
151 
152  /// @return Returns a hash value of the text.
153  HashedId get_text_id() const { return text_id_; }
154 
155  /// @return Returns the size value.
156  const mathfu::vec2i &get_size() const { return size_; }
157 
158  /// @return Returns the font size.
159  float get_font_size() const { return font_size_; }
160 
161  /// @return Returns a flag to indicate if the buffer has caret info.
162  bool get_caret_info_flag() const { return caret_info_; }
163 
164  private:
165  HashedId font_id_;
166  HashedId text_id_;
167  float font_size_;
168  mathfu::vec2i size_;
169  bool caret_info_;
170 };

462 /// @class FontMetrics
463 ///
464 /// @brief This class has additional properties for font metrics.
465 //
466 /// For details of font metrics, refer http://support.microsoft.com/kb/32667
467 /// In this class, ascender and descender values are retrieved from FreeType
468 /// font property.

479  ascender_(0),

481  external_leading_(0) {}
482 
483  /// The constructor with initialization parameters for FontMetrics.
484  FontMetrics(const int32_t base_line, const int32_t internal_leading,
485  const int32_t ascender, const int32_t descender,
486  const int32_t external_leading)
487  : base_line_(base_line),
488  internal_leading_(internal_leading),
489  ascender_(ascender),
490  descender_(descender),
491  external_leading_(external_leading) {
492  assert(internal_leading >= 0);
493  assert(ascender >= 0);
494  assert(descender <= 0);
495  assert(external_leading <= 0);
496  }
497 
498  /// The destructor for FontMetrics.
499  ~FontMetrics() {}
500 
501  /// @return Returns the baseline value as an int32_t.
502  int32_t base_line() const { return base_line_; }

505  ///

507  void set_base_line(const int32_t base_line) { base_line_ = base_line; }
508 
509  /// @return Returns the internal leading parameter as an int32_t.
510  int32_t internal_leading() const { return internal_leading_; }
511

513  ///
514  /// @param[in] internal_leading An int32_t to set as the internal

517  assert(internal_leading >= 0);
518  internal_leading_ = internal_leading;
519  }

521  /// @return Returns the ascender value as an int32_t.
522  int32_t ascender() const { return ascender_; }
523 
524  /// @brief Set the ascender value.
525  ///
526  /// @param[in] ascender An int32_t to set as the ascender value.
527  void set_ascender(const int32_t ascender) {
528  assert(ascender >= 0);
529  ascender_ = ascender;
530  }
531 
532  /// @return Returns the descender value as an int32_t.
533  int32_t descender() const { return descender_; }
534 
535  /// @brief Set the descender value.
536  ///
537  /// @param[in] descender An int32_t to set as the descender value.
538  void set_descender(const int32_t descender) {
539  assert(descender <= 0);
540  descender_ = descender;
541  }
542 
543  /// @return Returns the external leading parameter value as an int32_t.
544  int32_t external_leading() const { return external_leading_; }
545 
546  /// @brief Set the external leading parameter value.
547  ///
548  /// @param[in] external_leading An int32_t to set as the external
549  /// leading value.
550  void set_external_leading(const int32_t external_leading) {
551  assert(external_leading <= 0);
552  external_leading_ = external_leading;
553  }
554 
555  /// @return Returns the total height of the glyph.
556  int32_t total() const {
557  return internal_leading_ + ascender_ - descender_ - external_leading_;
558  }
559 
560  private:
561  // Baseline: Baseline of the glpyhs.
562  // When rendering multiple glyphs in the same horizontal group,
563  // baselines need be aligned.
564  // Most of glyphs fit within the area of ascender + descender.
565  // However some glyphs may render in internal/external leading area.
566  // (e.g. , underlines)
567  int32_t base_line_;
568 
569  // Internal leading: Positive value that describes the space above the
570  // ascender.
571  int32_t internal_leading_;
572 
573  // Ascender: Positive value that describes the size of the ascender above
574  // the baseline.
575  int32_t ascender_;
576 
577  // Descender: Negative value that describes the size of the descender below
578  // the baseline.
579  int32_t descender_;
580 
581  // External leading : Negative value that describes the space below
582  // the descender.
583  int32_t external_leading_;
584 };

655  /// @param[in] size A size of the FontBuffer in a number of glyphs.
656  /// @param[in] caret_info Indicates if the FontBuffer also maintains a caret
657  /// position buffer.
658  ///
659  /// @note Caret position does not match to glpyh position 1 to 1, because a
660  /// glyph can have multiple caret positions (e.g. Single 'ff' glyph can have 2
661  /// caret positions).

664  /// caret position information in the FontBuffer.

667  vertices_.reserve(size * kVerticesPerCodePoint);
